Matrices and their inverses

Sigma <- matrix(c(1.0, -0.7, -0.7, 1.0), nrow=2, ncol=2, byrow=TRUE)
Sigma_inv <- solve(Sigma)
test_inverse <- Sigma %*% Sigma_inv

Matrix Composition

A <- chol(Sigma)  #Cholesky decomposition of sigma
t(A)  #Transpose of a matrix
##      [,1]      [,2]
## [1,]  1.0 0.0000000
## [2,] -0.7 0.7141428
t(A) %*% A  #equals Sigma
##      [,1] [,2]
## [1,]  1.0 -0.7
## [2,] -0.7  1.0

Orthogonal Vectors and Matrices

x1 <- rnorm(10)
x2 <- rnorm(10)
x1 <- cbind(x1)
x2 <- cbind(x2)
X <- cbind(x1, x2)
(t(X) %*% X)/nrow(X)  #covariance matrix
##            x1         x2
## x1  0.8575328 -0.1140981
## x2 -0.1140981  0.8239374
w1 <- cbind(rnorm(10000))
w2 <- cbind(rnorm(10000))
W <- cbind(w1, w2)
W_cov <- (t(W) %*% W)/nrow(W)
W_cov
##              [,1]         [,2]
## [1,]  0.997980564 -0.007683724
## [2,] -0.007683724  1.003287539
